<script>
document.addEventListener('DOMContentLoaded', () => {

  /* ‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê  INITIALISE EVERY .dnd-ex  ‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê */
  document.querySelectorAll('.dnd-ex').forEach(ex=>{
      const ordered   = ex.dataset.order === 'true';
      const answerArr = ex.dataset.answer.split(',');            // ["2","5"] or ["2","4","3"]
      const choices   = ex.querySelector('.choices');
      const resetBtn  = ex.querySelector('.reset');
      const checkBtn  = ex.querySelector('.check');
      let dragging = null;

      /* ‚îÄ‚îÄ build slots w/ arrows for ordering tasks ‚îÄ‚îÄ */
      let targets;
      if (ordered){
        const slotsWrap = ex.querySelector('.slots');
        for (let i=0;i<answerArr.length;i++){
          const slot=document.createElement('div');
          slot.className='dnd-box slot'; slot.dataset.pos=i;
          slotsWrap.appendChild(slot);
          if (i<answerArr.length-1){
            const arrow=document.createElement('span');
            arrow.className='arrow'; arrow.innerHTML='&rarr;';
            slotsWrap.appendChild(arrow);
          }
        }
        targets = ex.querySelectorAll('.slot');
      }else{
        targets = [ ex.querySelector('.target') ];               // single target box
      }

      /* ‚îÄ‚îÄ make cards draggable ‚îÄ‚îÄ */
      ex.querySelectorAll('.dnd-item').forEach(card=>{
         card.addEventListener('dragstart', () => dragging = card);
      });

      /* ‚îÄ‚îÄ choices box (unlimited drops) ‚îÄ‚îÄ */
      makeDroppable(choices,false);

      /* ‚îÄ‚îÄ every target box ‚îÄ‚îÄ */
      targets.forEach(t => makeDroppable(t, ordered));           // single-card if ordered task

      function makeDroppable(box, single){
        box.addEventListener('dragover',e=>{
           if (!single || box.children.length===0){ e.preventDefault(); box.classList.add('dragover'); }
        });
        box.addEventListener('dragleave',()=>box.classList.remove('dragover'));
        box.addEventListener('drop',e=>{
           e.preventDefault(); box.classList.remove('dragover');
           if (!single || box.children.length===0) box.appendChild(dragging);
        });
      }

      /* ‚îÄ‚îÄ RESET ‚îÄ‚îÄ */
      resetBtn.addEventListener('click', reset);

      function reset(){
        targets.forEach(t=>{
          while(t.firstChild) choices.appendChild(t.firstChild);
          t.classList.remove('ok');
        });
      }

      /* ‚îÄ‚îÄ CHECK ‚îÄ‚îÄ */
      checkBtn.addEventListener('click', ()=>{
        let ok;
        if (ordered){
          const picked=[...targets].map(t=>t.firstElementChild?.dataset.key||'');
          ok = picked.join() === answerArr.join();
        }else{
          const picked=[...targets[0].querySelectorAll('.dnd-item')].map(x=>x.dataset.key).sort();
          ok = picked.length===answerArr.length &&
               picked.join() === answerArr.slice().sort().join();
        }

        if (ok){
          alert('‚úÖ Correct!');
          targets.forEach(t=>t.classList.add('ok'));
        }else{
          alert('‚ùå Not correct.');
          reset();                                              // auto-reset on wrong attempt
        }
      });
  });
});
</script>
