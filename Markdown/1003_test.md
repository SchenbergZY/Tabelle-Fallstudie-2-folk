<script>
document.addEventListener('DOMContentLoaded', () => {

  const ex       = document.querySelector('.dnd-ex');
  const choices  = ex.querySelector('.choices');
  const slots    = ex.querySelectorAll('.slot');
  const answer   = ex.dataset.ans.split(',');   // ["2","4","3"]
  let   dragging = null;

  /* make every card draggable */
  ex.querySelectorAll('.dnd-item').forEach(card =>
    card.addEventListener('dragstart', () => dragging = card)
  );

  /* choices box ‚Äì unlimited drops */
  makeDroppable(choices, false);

  /* each slot ‚Äì accept only one card */
  slots.forEach(slot => makeDroppable(slot, true));

  /* reusable drop helper */
  function makeDroppable(box, single){
    box.addEventListener('dragover',  e => {
      if (!single || box.children.length === 0){ e.preventDefault(); box.classList.add('dragover'); }
    });
    box.addEventListener('dragleave', () => box.classList.remove('dragover'));
    box.addEventListener('drop',      e => {
      e.preventDefault(); box.classList.remove('dragover');
      if (!single || box.children.length === 0) box.appendChild(dragging);
    });
  }

  /* CHECK button */
  ex.querySelector('.check').addEventListener('click', () => {
    const picked = Array.from(slots).map(s => s.firstElementChild?.dataset.key || '');
    const correct = picked.join() === answer.join();

    if (correct){
      alert('‚úÖ Correct!');
      slots.forEach(s => s.classList.add('ok'));
    } else {
      alert('‚ùå Not correct.');
      resetCards();                       // auto-reset on wrong attempt
    }
  });

  /* RESET button */
  ex.querySelector('.reset').addEventListener('click', resetCards);

  function resetCards(){
    slots.forEach(s => {
      if (s.firstElementChild) choices.appendChild(s.firstElementChild);
      s.classList.remove('ok');
    });
  }
});
</script>
